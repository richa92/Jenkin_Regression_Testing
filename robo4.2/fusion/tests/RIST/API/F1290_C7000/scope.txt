*** Keywords ***

Get Scope URI
	[Arguments]		${name}
	${resp} = 	Fusion Api Get Scope
	${count} =  Get From Dictionary  ${resp}  count
	Return from keyword if  ${count}==0  '/bad_scope_uri'
	:FOR    ${scope}    IN    @{resp['members']}
	\       Return From Keyword If  '${scope["name"]}' == '${name}'    ${scope["uri"]}
	\       Continue for loop
	[Return]	'/bad_scope_uri'

Get Server hardware
	[Arguments]		${name}
	${resp} = 	Fusion Api Get Server Hardware  param=?filter="'name'=='${name}'"
	Log to console and logfile  ${resp}  TRACE
	${count} =  Get From Dictionary  ${resp}  count
	Return from keyword if  ${count}==0  ${resp}
	[Return]	${resp['members'][0]}
