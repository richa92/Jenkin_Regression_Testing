	${lig_uri}	  Get Variable Value	${lig_edit['uri']}

	${Value}	Run Keyword If	  ${flag} == 0	  Create List	 ${Value}	 ELSE	 Set Variable	 ${Value}
	${len}	  Get Length	${Key}
	${temp}	   Copy Dictionary	  ${edit_lig_ntp}
	:FOR	${i}	IN RANGE	 0	  ${len}
	\	Set To Dictionary	 ${temp}	${Key[${i}]}	${Value[${i}]}
