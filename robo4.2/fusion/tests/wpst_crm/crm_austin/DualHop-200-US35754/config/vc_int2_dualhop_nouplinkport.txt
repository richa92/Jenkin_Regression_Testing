### Config file for 1st test case of dualhop feature
### Set up FCoE network without uplinkport associated with uplink set which will raise a critical issue.

set session Timeout=0

set snmp fc SmisEnabled=true

set domain MacType=Factory-Default
set domain WwnType=Factory-Default
set serverid Type=Factory-Default

add network net_1
add uplinkport enc0:1:x4 network=net_1
add uplinkport enc0:2:x4 network=net_1

add uplinkset up_A1
add network fcoe_101 -fcoe uplinkset=up_A1 vlanid=101
