*** Keywords ***

Set Variable With Zero
    [Arguments]     ${index}    ${var}
    ${scope_name} =       set variable    ${var}${0}${index}
    [Return]    ${scope_name}

Set Variable Without Zero
    [Arguments]     ${index}    ${var}
    ${scope_name} =       set variable    ${var}${index}
    [Return]    ${scope_name}
