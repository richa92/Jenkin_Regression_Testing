${blnExpected}		True


*** Test Cases ***
Delete storage volume in Fusion
	Log into Fusion appliance as Administrator
	${blnSelectvolume} =	Set Variable  False
	Set Global Variable			${blnSelectvolume}
	${blnDeletevolume} =	Fusion UI Delete Storage Volumes	${TestData.storagevolumes[3].name}	ApplianceAndStorage
	Run Keyword If		'${blnDeletevolume}' == '${blnExpected}'	Log		Storage volume deleted successfully.
	Run Keyword Unless		'${blnDeletevolume}' == '${blnExpected}'	Fail	Storage volume is not deleted.

Add storage volume in Fusion
	Log into Fusion appliance as Administrator
	${blnSelectvolume} =	Set Variable  False
	Set Global Variable			${blnSelectvolume} 
	${blnAddvolume} =	Fusion UI Create Storage Volumes	@{TestData.storagevolume}
	Run Keyword If		'${blnAddvolume}' == '${blnExpected}'	Log		Storage volume created successfully.
	Run Keyword Unless		'${blnAddvolume}' == '${blnExpected}'	Fail		Storage volume is not created.

	Log into Fusion appliance as Administrator

Add label to volume in Fusion
	Log into Fusion appliance as Administrator
	${blnLabelvolume} =	Set Variable	False
	Set Global Variable			${blnLabelvolume}
	${blnLabelvolume} =		Fusion UI add label to volume	@{Testdata.volumelabel}
	Run Keyword If		'${blnLabelvolume}' == '${blnExpected}'		Log		Label is added to storage volume
	Run Keyword Unless		'${blnLabelvolume}' == '${blnExpected}'		Fail		Failed to add Label to storage volume
