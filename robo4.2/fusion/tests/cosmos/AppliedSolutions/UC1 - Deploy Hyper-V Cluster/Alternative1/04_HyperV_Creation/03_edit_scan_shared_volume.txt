Edit storage volume
       Log into Fusion appliance as Administrator
       Fusion UI edit storage volumes    @{TestData.editstoragevolumes}

Rescan and increase the size of CSV
       Run		powershell Enable-PSRemoting -Force
       ${domainusername} =	Set Variable	${TestData.clusterNode[0].domainusername}
       ${domainpassword} =	Set Variable	${TestData.clusterNode[0].domainpassword}
       ${diskcapacity} =    Evaluate	${TestData.editstoragevolumes[0].capacity} - 1
       Run           powershell $Username = '${domainusername}'; $Password ='${domainpassword}'; $pass = ConvertTo-SecureString -AsPlainText $Password -Force; $Cred = New-Object System.Management.Automation.PSCredential -ArgumentList $Username,$pass; Invoke-Command -ComputerName '${dep_ip1}' -ScriptBlock { "rescan" | diskpart } -credential $Cred
       Run           powershell $Username = '${domainusername}'; $Password ='${domainpassword}'; $pass = ConvertTo-SecureString -AsPlainText $Password -Force; $Cred = New-Object System.Management.Automation.PSCredential -ArgumentList $Username,$pass; Invoke-Command -ComputerName '${dep_ip1}' -ScriptBlock { $num="Get-Disk |where operationalStatus -eq offline";Resize-Partition -DiskNumber $num.Number -PartitionNumber 1 -Size ${diskcapacity}GB } -credential $Cred
